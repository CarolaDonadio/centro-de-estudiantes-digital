💻 FUNCIONES DISPONIBLES:
═══════════════════════════════════════════════════════════════

1. filtrarNovedadesParaUsuario(novedades, usuario)
   → Retorna array de novedades visibles para el usuario

2. puedeVerNovedad(novedad, usuario)
   → Boolean: ¿Puede ver esta novedad?

3. puedeCrearNovedad(usuario, tipoNovedad)
   → Boolean: ¿Puede crear este tipo?

4. validarNovedadCreada(novedad, usuario)
   → { valido: Boolean, errores: String[] }

5. ordenarNovedades(novedades)
   → Array ordenado (destacadas primero, por fecha)

6. generarHTMLNovedad(novedad, mostrarAcciones)
   → String (HTML de tarjeta de novedad)

7. buscarNovedades(novedades, termino)
   → Array de resultados de búsqueda

8. agruparPorCategoria(novedades)
   → { categoriaId: [novedades...] }

📝 EJEMPLO RÁPIDO DE USO:
═══════════════════════════════════════════════════════════════

import { 
  filtrarNovedadesParaUsuario, 
  validarNovedadCreada 
} from './novedades-filtro.js';

// Usuario actual
const usuario = {
  rol: 'alumno',
  carrera_id: 3,
  materias: [1, 5, 7]
};

// 1. Cargar novedades
const respuesta = await fetch('/json/novedades.json');
const { novedades } = await respuesta.json();

// 2. Filtrar para este usuario
const visibles = filtrarNovedadesParaUsuario(novedades, usuario);
console.log(`${usuario.nombre} puede ver ${visibles.length} novedades`);

// 3. Validar antes de crear
const novedad = {
  titulo: 'Nueva noticia',
  contenido: 'Contenido...',
  puede_crear: ['alumno']
};

const validacion = validarNovedadCreada(novedad, usuario);
if (validacion.valido) {
  console.log('✅ Puede crear');
  // Guardar...
} else {
  console.log('❌ Errores:', validacion.errores);
}
